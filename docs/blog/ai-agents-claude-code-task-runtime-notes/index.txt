func (r *TaskRegistry) Register(task Task, state TaskStateBase) {
  r.mu.Lock()
  defer r.mu.Unlock()
  r.tasks[state.ID] = &taskEntry{Task: task, State: state}
  emitTaskStartedNotification(state)

func (r *TaskRegistry) Stop(ctx context.Context, taskID string) error {
  r.mu.RLock()
  entry, ok := r.tasks[taskID]
  r.mu.RUnlock()
  if !ok {
      return nil
  }
  if err := entry.Task.Kill(ctx); err != nil {
      return err
  }
  r.mu.Lock()
  entry.State.Status = TaskStatusStopped
  r.mu.Unlock()
  return nil
}
