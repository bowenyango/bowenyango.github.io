func DedupBySignature(configs []MCPServerConfig) []MCPServerConfig {
  seen := map[string]bool{}
  var out []MCPServerConfig
  for _, cfg := range configs {
      sig := ServerSignature(cfg)
      if !seen[sig] {
          seen[sig] = true
          out = append(out, cfg)
      }
  }
  return out
}
