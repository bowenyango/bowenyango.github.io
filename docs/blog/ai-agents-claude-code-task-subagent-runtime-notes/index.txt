func NewAgentRuntime(ctx context.Context, agentID string, def AgentDef, parent *ToolUseContext) (*AgentRuntime, error) {
  // Build agent-scoped context
  agentCtx := &ToolUseContext{
      Options:           buildAgentOptions(def, parent),
      AbortController:   linkAbortController(parent.AbortController),
      ReadFileState:     resolveFileState(def, parent),
      GetAppState:       buildAgentStateAccessor(agentID),
      SetAppState:       resolveSetAppState(def, parent),
      SetAppStateForTasks: parent.SetAppStateForTasks,
  }

  // Preload capabilities
  if err := preloadSkills(ctx, def.Skills, agentCtx); err != nil {
      return nil, err
  }
  mcp, err := connectAgentMCPServers(ctx, def.MCPServers, agentCtx)
  if err != nil {
      return nil, err
  }
  if err := runSubagentStartHooks(ctx, agentCtx); err != nil {
      mcp.Close()
      return nil, err
  }

  return &AgentRuntime{Context: agentCtx, MCP: mcp, AgentID: agentID}, nil
}

func (r *AgentRuntime) Close(ctx context.Context) {
  r.MCP.Close()
  teardownHooks(ctx, r.Context)
  cleanupChildShellTasks(ctx, r.AgentID, r.Context)
  recordSidechainTranscript(ctx, r.AgentID, r.Context)
}
