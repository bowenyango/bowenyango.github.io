func RunCoordinatorTurn(ctx context.Context, task string, backend SwarmBackend) (string, error) {
  plan, err := PlanWithCoordinator(ctx, task)
  if err != nil {
      return "", err
  }

  // Spawn all subagents in parallel
  ids := make([]string, len(plan.Subtasks))
  for i, subtask := range plan.Subtasks {
      id, err := backend.Spawn(ctx, subtask)
      if err != nil {
          return "", err
      }
      ids[i] = id
  }

  // Collect results
  results := make([]*SubagentResult, len(ids))
  var wg sync.WaitGroup

      wg.Add(1)

          results[i] = result
