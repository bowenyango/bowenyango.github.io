func (g *QueryGuard) Reserve() bool {
  g.mu.Lock()
  defer g.mu.Unlock()
  if g.state != QueryIdle {
      return false
  }
  g.state = QueryReserved
  go g.notify() // async notify — don't hold lock during callbacks
  return true
}

// TryStart: atomic transition reserved → running
func (g *QueryGuard) TryStart() bool {
  g.mu.Lock()
  defer g.mu.Unlock()
  if g.state != QueryReserved {
      return false
  }
  g.state = QueryRunning
  go g.notify()
  return true
}

func (g *QueryGuard) End() {
  g.mu.Lock()
  g.state = QueryIdle
  g.mu.Unlock()
  go g.notify()
}

func (g *QueryGuard) notify() {
  g.mu.Lock()
  listeners := append([]func(){}, g.listeners...)
  g.mu.Unlock()
  for _, l := range listeners {
      l()
  }
}
