  BehaviorAllow       Behavior = "allow"
  BehaviorAsk         Behavior = "ask"
  BehaviorDeny        Behavior = "deny"
  BehaviorPassthrough Behavior = "passthrough"
)

type ReasonType string

const (
  ReasonRule        ReasonType = "rule"
  ReasonMode        ReasonType = "mode"
  ReasonSafetyCheck ReasonType = "safetyCheck"
  ReasonClassifier  ReasonType = "classifier"
)

type Decision struct {
  Behavior     Behavior
  Reason       ReasonType
  Rule         *PermissionRule
  Message      string
  UpdatedInput map[string]any
}

func CheckToolPermission(ctx context.Context, tool Tool, input map[string]any, uc *ToolUseContext) (Decision, error) {
  if ctx.Err() != nil {
      return Decision{}, ctx.Err()
  }

  state := uc.GetAppState()
  perm := state.ToolPermissionContext

  // 1a. Tool-level deny rule
  if rule := GetDenyRule(perm, tool); rule != nil {
      return Decision{Behavior: BehaviorDeny, Reason: ReasonRule, Rule: rule}, nil
  }

  // 1b. Tool-level ask rule (unless sandboxed bash)
  if rule := GetAskRule(perm, tool); rule != nil && !CanSandboxAutoAllow(tool, input) {
      return Decision{Behavior: BehaviorAsk, Reason: ReasonRule, Rule: rule}, nil
  }

  // 1c. Tool-specific check
  toolResult, err := tool.CheckPermissions(ctx, tool.ParseInput(input), uc)
  if err != nil {
      return Decision{}, err
  }

  // 1d. Tool denied
  if toolResult.Behavior == BehaviorDeny {
      return toolResult, nil
  }

  // 1e. Requires interactive approval — cannot bypass
  if tool.RequiresUserInteraction() && toolResult.Behavior == BehaviorAsk {
      return toolResult, nil
  }

  // 1f. Content-specific ask rule — bypass-resistant
  if toolResult.Behavior == BehaviorAsk && toolResult.Reason == ReasonRule {
      return toolResult, nil
  }

  // 1g. Safety check — bypass-immune (.git, .claude, shell configs)
  if toolResult.Behavior == BehaviorAsk && toolResult.Reason == ReasonSafetyCheck {
      return toolResult, nil
  }

  // 2a. Bypass mode (re-read state — may have changed during async checks)
  state = uc.GetAppState()
  if IsBypassMode(state.ToolPermissionContext) {
      return Decision{Behavior: BehaviorAllow, Reason: ReasonMode}, nil
  }

  // 2b. Always-allow rule
  if rule := GetAlwaysAllowRule(perm, tool); rule != nil {
      return Decision{Behavior: BehaviorAllow, Reason: ReasonRule, Rule: rule}, nil
  }

  // 3. Passthrough → ask
  if toolResult.Behavior == BehaviorPassthrough {
      return Decision{Behavior: BehaviorAsk, Message: BuildRequestMessage(tool.Name())}, nil
  }
  return toolResult, nil
