class ExternalStore<T> {
private state: T
private listeners = new Set<() => void>()

constructor(initial: T) { this.state = initial }

getSnapshot(): T { return this.state }

setState(update: (prev: T) => T): void {
  this.state = update(this.state)
  this.listeners.forEach(l => l())
}

subscribe(listener: () => void): () => void {
  this.listeners.add(listener)
  return () => this.listeners.delete(listener)
}
}

class ExternalStore(Generic[T]):
  """Minimal external store: state + subscribers, no framework dependency."""

  def __init__(self, initial: T) -> None:
      self._state = initial
      self._listeners: set[Callable[[], None]] = set()
      self._lock = Lock()

  def get_snapshot(self) -> T:
      return self._state

  def set_state(self, update: Callable[[T], T]) -> None:
      with self._lock:
          self._state = update(self._state)
      # Notify outside the lock
      for listener in list(self._listeners):
          listener()

  def subscribe(self, listener: Callable[[], None]) -> Callable[[], None]:
      self._listeners.add(listener)
      return lambda: self._listeners.discard(listener)
